public class coupon {
	public static void main(String[] args) {
int n=Integer.parseInt(args[0]);
int count=collect(n);
System.out.println(count);
		
	}
	public static int getCoupon(int n)
	{
		return(int) (Math.random()*n);
		
	}
	public static int collect(int n)
	{
		boolean[] isCollected=new boolean[n];
		int count=0;
		int distinct=0;
		while (distinct<n) {
			 int value=getCoupon(n);
		count++;
		if(!isCollected[value])
		{
			distinct++;
			isCollected[value]=true;
		}}
		return count;
	
	

}
}

public class binarySearchString {

		static String[] a = { "AAA", "BBB", "CCC", "DDD", "EEE", "FFF", "GGG" };
		public static void main(String[] args) {
			System.out.println("Key Found at : " + binarySearch(a,"AAA") + " position");
		}
		public static int binarySearch(String arr[],String key)
		{
			int start=0;
			int end=arr.length-1;
			while(start<=end)
			{
				int mid=(start+end)/2;
				if(key==arr[mid])
				{
					return mid;
					}
				else if(key.compareTo(arr[mid])<0)
				{
					end=mid-1;
				}
				else
				{
					start=mid+1;
				}
			}
			return -1;
		}
	}

{System.out.println("Not an Anagram");}

static boolean areAnagram(char[]str1,char[]str2)
{
	int s1=str1.length;
	int s2=str2.length;
	if(s1!=s2)
	return false;
	Arrays.sort(str1);
	Arrays.sort(str2);
	for(int i=0;i<s1;i++)
	{
		if(str1[i]!=str2[i])
		{
			return false;
		}
		return true;
	}
	return false;
}

import java.util.Stack;

public class stack {

	public static void main(String[] args) {
		String s = "(5+6)∗(7+8)/(4+3)(5+6)∗(7+8)/(4+3)";
		String result = (stringsCheck(s)) ? "Balanced" : "Not Balanced";
		System.out.println(result);
	}
static boolean stringsCheck(String s)
{
	Stack <Character>st=new Stack<>();
	 for(int i=0;i<s.length();i++)
	 {
		 char c=s.charAt(i);
		 if(c=='(')
		 {
			 st.push(c);
		 }
		 else if(c==')') {
			 st.pop();
		 }
		 
	 }	
	 return st.isEmpty();
}
}
